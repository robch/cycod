USE ANTHROPIC

  AUTHENTICATION

    EXAMPLE 1: Set Anthropic API key via config

      cycod config set Anthropic.ApiKey KEY
      cycod config set Anthropic.ApiKey KEY --local
      cycod config set Anthropic.ApiKey KEY --user
      cycod config set Anthropic.ApiKey KEY --global

    EXAMPLE 2: Set Anthropic API key via environment variable

      # Bash (Linux/macOS)
      export ANTHROPIC_API_KEY="YOUR_API_KEY"

      # CMD (Windows)
      set ANTHROPIC_API_KEY=YOUR_API_KEY

      # PowerShell (Windows)
      $env:ANTHROPIC_API_KEY = "YOUR_API_KEY"

    EXAMPLE 3: Supply Anthropic API key via command line

      cycod chat --anthropic-api-key KEY

  MODEL SELECTION

    EXAMPLE 1: Set Anthropic model name via config

      cycod config set Anthropic.ModelName claude-sonnet-4
      cycod config set Anthropic.ModelName claude-sonnet-4 --local
      cycod config set Anthropic.ModelName claude-sonnet-4 --user
      cycod config set Anthropic.ModelName claude-sonnet-4 --global

    EXAMPLE 2: Set Anthropic model name via environment variable

      # Bash (Linux/macOS)
      export ANTHROPIC_MODEL_NAME="claude-sonnet-4"

      # CMD (Windows)
      set ANTHROPIC_MODEL_NAME=claude-sonnet-4

      # PowerShell (Windows)
      $env:ANTHROPIC_MODEL_NAME = "claude-sonnet-4"

    EXAMPLE 3: Use Claude Sonnet 4 for advanced reasoning tasks

      cycod chat --anthropic-model-name claude-sonnet-4
