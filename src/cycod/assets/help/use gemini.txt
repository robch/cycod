USE GEMINI

  AUTHENTICATION

    EXAMPLE 1: Set Google Gemini credentials via config
    
      cycod config set Google.Gemini.ApiKey API_KEY
      cycod config set Google.Gemini.ApiKey API_KEY --local
      cycod config set Google.Gemini.ApiKey API_KEY --user
      cycod config set Google.Gemini.ApiKey API_KEY --global

    EXAMPLE 2: Set Google Gemini credentials via environment variables
    
      # Bash (Linux/macOS)
      export GOOGLE_GEMINI_API_KEY="YOUR_API_KEY"

      # CMD (Windows)
      set GOOGLE_GEMINI_API_KEY=YOUR_API_KEY

      # PowerShell (Windows)
      $env:GOOGLE_GEMINI_API_KEY = "YOUR_API_KEY"
    
    EXAMPLE 3: Supply Google Gemini credentials via command line
    
      cycod chat --google-gemini-api-key API_KEY

  MODEL SELECTION

    EXAMPLE 1: Set Google Gemini model via config
    
      cycod config set Google.Gemini.ModelId gemini-2.5-pro
      cycod config set Google.Gemini.ModelId gemini-2.5-pro --local
      cycod config set Google.Gemini.ModelId gemini-2.5-pro --user
      cycod config set Google.Gemini.ModelId gemini-2.5-pro --global
    
    EXAMPLE 2: Set Google Gemini model via environment variable
    
      # Bash (Linux/macOS)
      export GOOGLE_GEMINI_MODEL_ID="gemini-2.5-pro"

      # CMD (Windows)
      set GOOGLE_GEMINI_MODEL_ID=gemini-2.5-pro

      # PowerShell (Windows)
      $env:GOOGLE_GEMINI_MODEL_ID = "gemini-2.5-pro"
    
    EXAMPLE 3: Use the latest Gemini model for development tasks

      cycod chat --google-gemini-model-id gemini-2.5-pro
