USE OPENAI

  AUTHENTICATION

    EXAMPLE 1: Set OpenAI API key via config

      cycod config set OpenAI.ApiKey KEY
      cycod config set OpenAI.ApiKey KEY --local
      cycod config set OpenAI.ApiKey KEY --user
      cycod config set OpenAI.ApiKey KEY --global

    EXAMPLE 2: Set OpenAI API key via environment variable

      # Bash (Linux/macOS)
      export OPENAI_API_KEY="YOUR_API_KEY"

      # CMD (Windows)
      set OPENAI_API_KEY=YOUR_API_KEY

      # PowerShell (Windows)
      $env:OPENAI_API_KEY = "YOUR_API_KEY"

    EXAMPLE 3: Supply OpenAI API key via command line

      cycod chat --openai-api-key KEY

  MODEL SELECTION

    EXAMPLE 1: Set OpenAI model name via config

      cycod config set OpenAI.ChatModelName gpt-5-mini
      cycod config set OpenAI.ChatModelName gpt-5-mini --local
      cycod config set OpenAI.ChatModelName gpt-5-mini --user
      cycod config set OpenAI.ChatModelName gpt-5-mini --global

    EXAMPLE 2: Set OpenAI model name via environment variable

      # Bash (Linux/macOS)
      export OPENAI_CHAT_MODEL_NAME="gpt-5-mini"

      # CMD (Windows)
      set OPENAI_CHAT_MODEL_NAME=gpt-5-mini

      # PowerShell (Windows)
      $env:OPENAI_CHAT_MODEL_NAME = "gpt-5-mini"

    EXAMPLE 3: Use a specific model optimized for code generation

      cycod chat --openai-chat-model-name gpt-5-mini
